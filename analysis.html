Python Code

The following Python code loads the employee dataset, calculates the frequency count for the "HR" department, prints it, and creates a histogram of department distribution using seaborn/matplotlib.


import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Example employee dataset (100 employees)
data = {
    "EmployeeID": range(1, 101),
    "Department": (
        ["HR"] * 18 +
        ["Finance"] * 20 +
        ["Engineering"] * 22 +
        ["Sales"] * 20 +
        ["Marketing"] * 20
    ),
    "Region": ["North", "South", "East", "West", "Central"] * 20,
    "PerformanceScore": [60, 72, 88, 91, 75] * 20,
}

df = pd.DataFrame(data)

# 1. Calculate frequency count for the "HR" department
hr_count = (df["Department"] == "HR").sum()
print("Frequency count for the \"HR\" department:", hr_count)

# 2. Create a histogram (count plot) of department distribution
plt.figure(figsize=(8, 6))
sns.countplot(data=df, x="Department")
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Number of Employees")
plt.tight_layout()

# 3. Save the visualization
plt.savefig("departments_histogram.png")
plt.close()
