import pandas as pd import matplotlib.pyplot as plt import seaborn as sns # --- Dataset with EXACTLY 18 HR rows --- departments = ( ["HR"] * 18 + ["Finance"] * 20 + ["Engineering"] * 22 + ["Sales"] * 20 + ["Marketing"] * 20 ) data = { "EmployeeID": range(1, 101), "Department": departments, "Region": ["North","South","East","West","Central"] * 20, "PerformanceScore": [60,72,88,91,75] * 20 } df = pd.DataFrame(data) # --- HR frequency (must be 18) --- hr_count = (df["Department"] == "HR").sum() print("Frequency count for HR department:", hr_count) # --- Histogram of department distribution --- plt.figure(figsize=(6,4)) sns.countplot(data=df, x="Department", palette="coolwarm") plt.title("Department Distribution") plt.xlabel("Department") plt.ylabel("Count") plt.tight_layout() plt.show()
