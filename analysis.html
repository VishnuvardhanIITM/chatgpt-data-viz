import pandas as pd import matplotlib.pyplot as plt import seaborn as sns # --- Synthetic Dataset of Employees (100 rows) --- data = { "EmployeeID": range(1, 101), "Department": [ "HR","Finance","Engineering","Sales","Marketing" ] * 20, # repeating "Region": ["North","South","East","West","Central"] * 20, "PerformanceScore": [60,72,88,91,75] * 20 } df = pd.DataFrame(data) # --- Frequency count for HR department --- hr_count = (df["Department"] == "HR").sum() print("Frequency count for HR department:", hr_count) # --- Histogram of department distribution --- plt.figure(figsize=(6,4)) sns.countplot(data=df, x="Department", palette="viridis") plt.title("Department Distribution") plt.xlabel("Department") plt.ylabel("Count") plt.tight_layout() plt.show()
